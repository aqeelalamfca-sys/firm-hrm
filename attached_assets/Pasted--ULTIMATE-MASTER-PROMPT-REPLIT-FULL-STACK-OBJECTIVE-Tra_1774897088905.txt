🔷 FUNCTION: AUTO UPDATE GENERATOR
async function generateUpdate(topic) {
  const res = await openai.chat.completions.create({
    model: "gpt-5",
    messages: [
      {
        role: "system",
        content: "You are a Pakistan regulatory expert (FBR, SECP, PSX, SBP). Generate short professional update for clients."
      },
      {
        role: "user",
        content: topic
      }
    ]
  });

  return res.choices[0].message.content;
}
