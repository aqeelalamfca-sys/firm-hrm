CODE:

import OpenAI from "openai";

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

export async function generateRegulatoryUpdates() {
  const prompt = `
  Generate latest regulatory updates for Pakistan:
  - FBR (tax, SROs, ATL, deadlines)
  - SECP (Companies Act, filings)
  - SBP (banking, forex)
  - PSX (market, compliance)

  Format:
  - Title
  - Authority
  - Summary (2–3 lines)
  - Effective Date
  - Priority: High/Medium/Low
  - Category: Tax / Corporate / Financial / Compliance

  Keep professional for Chartered Accountancy firm.
  Max 5 updates.
  `;

  const response = await client.chat.completions.create({
    model: "gpt-5.3",
    messages: [{ role: "user", content: prompt }],
  });

  return response.choices[0].message.content;
}

PHASE 2 — BACKEND API ENDPOINT
--------------------------------------------------

Create:

/server/routes/ai.routes.ts

import express from "express";
import { generateRegulatoryUpdates } from "../services/ai.service";

const router = express.Router();

router.get("/generate-updates", async (req, res) => {
  try {
    const data = await generateRegulatoryUpdates();
    res.json({ success: true, data });
  } catch (err) {
    res.status(500).json({ success: false, message: err.message });
  }
});

export default router;
